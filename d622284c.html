html = """

<body>
<div id="songs-list'>
<h2 class="title">经典老歌</h2>
<p class="introduction">
    经典老歌列表
</p>
<ul id="list" class="list-group">
    <li data-view="2">1一路上有你</li>
    <li data-view="7">2沧海一声笑
        <a href="/2.mp3" singer="任贤齐"></a>

    <li data-view="4" class="active">3往事随风
